// Функция для отображения страницы
function showPage(pageNumber) {
    document.getElementById('page1').classList.remove('active');
    document.getElementById('page2').classList.remove('active');
    document.getElementById('page' + pageNumber).classList.add('active');
}

// Функция для отображения содержимого билета
function showTicket(ticketNumber) {
    const ticket = tickets.find(t => t.number === ticketNumber);
    if (ticket) {
        document.getElementById('ticketPageTitle').textContent = `Билет №${ticket.number}: ${ticket.title}`;
        document.getElementById('ticketContent').innerHTML = ticket.content;
        
        // Добавляем раздел с иллюстрациями
        const illustrationsSection = document.getElementById('illustrationsSection');
        if (ticket.illustrations && ticket.illustrations.length > 0) {
            let illustrationsHTML = '<h3>Иллюстрации по теме:</h3><div class="illustration-links">';
            ticket.illustrations.forEach(illustration => {
                illustrationsHTML += `<a href="${illustration.url}" target="_blank" class="illustration-link">${illustration.name}</a>`;
            });
            illustrationsHTML += '</div>';
            illustrationsSection.innerHTML = illustrationsHTML;
        } else {
            illustrationsSection.innerHTML = '';
        }
        
        showPage(2);
    }
}

// Инициализация списка билетов
function initTicketList() {
    const ticketList = document.getElementById('ticketList');
    ticketList.innerHTML = '';
    
    tickets.forEach(ticket => {
        const ticketItem = document.createElement('div');
        ticketItem.className = 'ticket-item';
        ticketItem.onclick = () => showTicket(ticket.number);
        
        ticketItem.innerHTML = `
            <div class="ticket-number">Билет №${ticket.number}</div>
            <div class="ticket-title">${ticket.title}</div>
        `;
        
        ticketList.appendChild(ticketItem);
    });
}

// Инициализация при загрузке страницы
window.onload = function() {
    initTicketList();
};
